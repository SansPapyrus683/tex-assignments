                  \left|\frac{e^{3z}}{1+e^z}\right|

                   & \le \frac{e^{3R}}{\left|1-e^R\right|}             \\

            \end{align*}

            \begin{align*}
                  |\Log z|
                   & = |\log |z| + i\arg z|      \\
                   & = |i\arg z|                 \\
                   & = \arg z                    \\
                   & < \frac{\pi}{2}\quad\square
            \end{align*}

            Since $z=ai$ where $a \in [0, 1]$,
            \begin{align*}
                  |\exp(\sin z)|

                   & = \left|\exp\left(\frac{e^{-a}-e^{a}}{2i}\right)\right|                          \\
